	f.nextSibling
	function next(elem){
		do{
			elem = elem.nextSibling;
		}while(elem && elem.nodeType != 1);
		return elem;
	}
	function firstchild(){
		elem = elem.firstchild;
		return elem && (elem.nodeType == 1?elem:next(elem));
	}
